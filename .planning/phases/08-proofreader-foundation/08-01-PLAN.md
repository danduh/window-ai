From RESEARCH.md §1 (Proofreader API draft spec, observed Chrome 141–145 OT behavior):

```typescript
// Spec union (6 values). Chrome OT reliably returned only "spelling" + "grammar";
// declare all 6 so Phase 9 rendering does not need a type change.
type ProofreaderCorrectionType =
  | 'spelling'
  | 'punctuation'
  | 'capitalization'
  | 'preposition'
  | 'missing-words'
  | 'grammar';

interface ProofreaderCorrection {
  startIndex: number;            // byte offset in original input, inclusive
  endIndex: number;              // byte offset, exclusive
  correction: string;            // replacement text
  types?: ProofreaderCorrectionType[];  // present only when includeCorrectionTypes:true
  explanation?: string;          // present only when includeCorrectionExplanations:true
}

interface ProofreadResult {
  correctedInput: string;
  corrections: ProofreaderCorrection[];  // ordered ascending by startIndex
}

interface ProofreaderCreateOptions {
  includeCorrectionTypes?: boolean;
  includeCorrectionExplanations?: boolean;
  correctionExplanationLanguage?: string;   // BCP-47
  expectedInputLanguages?: string[];        // BCP-47
  signal?: AbortSignal;
  monitor?: (m: AICreateMonitor) => void;
}

interface ProofreaderProofreadOptions {
  signal?: AbortSignal;
}

interface Proofreader {
  proofread(input: string, options?: ProofreaderProofreadOptions): Promise<ProofreadResult>;
  destroy(): void;
  readonly includeCorrectionTypes: boolean;
  readonly includeCorrectionExplanations: boolean;
  readonly expectedInputLanguages: ReadonlyArray<string> | null;
  readonly correctionExplanationLanguage: string | null;
}

interface ProofreaderConstructor {
  create(options?: ProofreaderCreateOptions): Promise<Proofreader>;
  availability(options?: { expectedInputLanguages?: string[] }): Promise<'available' | 'downloadable' | 'downloading' | 'unavailable'>;
}
```
