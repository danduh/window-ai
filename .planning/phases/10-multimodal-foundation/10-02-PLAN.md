    Send handler (10-PATTERNS.md §MultimodalChatPanel.tsx + Pattern 3 streaming):
    ```
    const handleSend = async () => {
      if (!pendingImage || !text.trim() || pageState !== 'ready') return;

      // 1. Create object URL at COMMIT TIME (not for the pending preview — Pitfall 2)
      const objectUrl = URL.createObjectURL(pendingImage);
      objectUrlSetRef.current.add(objectUrl);

      // 2. Append user + empty assistant messages
      const streamingId = crypto.randomUUID();
      const userText = text.trim();
      const userImage = pendingImage;
      setMessages(prev => [
        ...prev,
        { id: crypto.randomUUID(), role: 'user', text: userText, attachedImageUrl: objectUrl },
        { id: streamingId, role: 'assistant', text: '' },
      ]);

      // 3. Reset input + transition state
      setText('');
      setPendingImage(null);
      setPageState('prompting');

      // 4. Create abort controller for this stream
      abortControllerRef.current = new AbortController();

      // 5. Call service + accumulate chunks
      try {
        const stream = await promptWithImage(userText, userImage, { signal: abortControllerRef.current.signal });
        const reader = stream.getReader();
        try {
          while (true) {
            const { value, done } = await reader.read();
            if (done) break;
            setMessages(prev => prev.map(m => m.id === streamingId ? { ...m, text: m.text + value } : m));
          }
          setPageState('ready');
        } finally {
          reader.releaseLock();
        }
      } catch (err) {
        const message = err instanceof Error ? err.message : 'Unknown error';
        setMessages(prev => prev.map(m => m.id === streamingId
          ? { ...m, text: '', error: `Couldn't process image — ${message}` }
          : m));
        setPageState('ready'); // return to ready so the user can retry or send a new message
      }
    };
    ```
